package main

import (
	"log"
	"net"
)

func main() {
	ifs, err := net.Interfaces()
	log.Println("Looking up ips...")
	if err != nil {
		log.Fatal(err)
	}

	for _, ief := range ifs {
		log.Println(ief.Name)
		addrs, err := ief.Addrs()

		if err != nil {
			log.Fatal(err)
		}

		for _, addr := range addrs {
			// if addr.isLookback
			log.Println("- " + addr.String())
		}
	}
	log.Println("Done.")
}
